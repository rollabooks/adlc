\begin{center}

\toprule

\midrule

\texttt{Ctrl+Shift+I} (Win/Linux) / \texttt{Cmd+Shift+I} (Mac) & Open Copilot Chat (Agent Mode) \\
\texttt{Ctrl+I} (Win/Linux) / \texttt{Cmd+I} (Mac) & Copilot Inline Chat \\
\texttt{Tab} & Accept Copilot suggestion \\
\texttt{Esc} & Dismiss Copilot suggestion \\
\texttt{Ctrl+`} & Open/close the integrated terminal \\
\texttt{Ctrl+Shift+E} & Explorer (files) \\
\texttt{Ctrl+Shift+G} & Source Control (Git) \\
\texttt{Ctrl+P} & Quick Open (find file) \\
\texttt{Ctrl+Shift+F} & Search across all files \\
\bottomrule
\end{tabularx}

\section{Node.js and npm}

\begin{center}
\begin{tabularx}{\textwidth}{l X}
\toprule
\textbf{Command} & \textbf{Description} \\
\midrule
\texttt{npm init -y} & Create \texttt{package.json} \\
\texttt{npm install <package>} & Install a dependency \\
\texttt{npm install -D <package>} & Install a dev dependency \\
\texttt{npm run dev} & Start in development mode \\
\texttt{npm run build} & Production build \\
\texttt{npm audit} & Check for vulnerabilities \\
\texttt{npm audit fix} & Apply automatic vulnerability fixes \\
\texttt{npx <command>} & Run a package without global installation \\
